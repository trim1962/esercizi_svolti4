	2x^2+2x+7>&0\\
	2x^2+2x+7=&0\\
	x_{1,2}=&\xunodue{2}{+2}{7}=\dfrac{-2\pm\sqrt{4-32}}{4}\\
	=&\dfrac{-2\pm\sqrt{-28}}{4}
	\qquad\text{nessuna soluzione}\\
	x^2>&0\\
	x^2=&0\\
	x_{1,2}=&\xunodue{1}{0}{0}=\dfrac{0\pm\sqrt{0-0}}{2}\\
	=&\dfrac{0\pm\sqrt{0}}{2}=\dfrac{0}{2}=0\\
	\end{align*}
