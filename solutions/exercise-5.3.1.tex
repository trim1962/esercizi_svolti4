	\begin{align*}
	2x+1>&0\\
	2x>&-1\\
	x>&-\dfrac{1}{2}\\
	x_{1,2}=&\xunodue{6}{+13}{+6}=\dfrac{-13\pm\sqrt{169-154}}{12}\\
	=&\dfrac{-13\pm\sqrt{25}}{12}=\dfrac{-13\pm 5}{12}\\
	=&\begin{cases}
	x_1=-\dfrac{18}{12}=-\dfrac{3}{2}\\
	x_2=-\dfrac{8}{12}=-\dfrac{2}{3}
	\end{cases}\\
	\end{align*}
